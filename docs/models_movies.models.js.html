const getAllMovie = async () => {
  let client;
  try {
    const pool = bdConnect();
    client = await pool.connect();

    const answer = await client.query(movies.getAllMovies);

    return answer.rows;
  } catch (error) {
    throw error;
  } finally {
    client.release();
  }
};
